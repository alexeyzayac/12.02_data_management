# 1. Поднимите чистый инстанс MySQL версии 8.0+
1. sudo apt update
2. sudo apt install -y mysql-server
3. sudo systemctl enable --now mysql

# 2. Создайте учётную запись sys_temp
4. sudo mysql
5. CREATE USER 'sys_temp'@'localhost' IDENTIFIED BY '123';
6. ALTER USER 'sys_temp'@'localhost' IDENTIFIED WITH mysql_native_password BY '123';

# 3. Выполните запрос на получение списка пользователей в базе данных *
7. SELECT user FROM mysql.user;


# 4. Дайте все права для пользователя sys_temp
8. GRANT ALL PRIVILEGES ON *.* TO 'sys_temp'@'localhost';
9. FLUSH PRIVILEGES;

# 5. Выполните запрос на получение списка прав для пользователя sys_temp *
10. SHOW GRANTS FOR 'sys_temp'@'localhost';
11. EXIT;

# 6. Переподключитесь к базе данных от имени sys_temp.
12. mysql -u sys_temp -p
13. SELECT USER();
14. EXIT;

# 7. По ссылке https://downloads.mysql.com/docs/sakila-db.zip скачайте дамп базы данных
15. mkdir ~/netology
16. cd ~/netology
17. wget https://downloads.mysql.com/docs/sakila-db.zip
18. unzip sakila-db.zip 
19. cd sakila-db

# 8. Восстановите дамп в базу данных
20. mysql -u sys_temp -p < sakila-schema.sql
21. mysql -u sys_temp -p < sakila-data.sql

#9. При работе в командной строке используйте команду для получения всех таблиц базы данных
22. mysql -u sys_temp -p
23. USE sakila;
24. SHOW TABLES;
